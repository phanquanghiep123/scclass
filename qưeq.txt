select (select (sum(tbl4.qty_view) ) AS qty_view  from members AS tbl1
     join article as tbl2 on tbl2.member_id = tbl1.id
     join radios as tbl3 on tbl3.article_id = tbl2.id
     join common_tracking as tbl4 on tbl4.reference_id = tbl2.id and tbl4.type_object = "radio"
     where tbl1.id = tbl4.article_id
) AS ,tbl1.* from (select tbl1.* from common_follow AS tbl1  where tbl1.member_id = $user_id and type_object = 'station' and status = 1)
        AS tbl1 join members as tbl2 on tbl1.reference_id = tbl2.id
        join article as tbl3 on tbl2.id = tbl3.member_id
        join (
                select tbl1.* 
                from radios AS tbl1 
                where tbl1.article_id !=0
                group by tbl1.article_id 
                order by tbl1.id DESC 
        ) AS tbl4 on tbl4.article_id = tbl3.id
        join
        group by tbl1.reference_id
        order by tbl2.id  DESC 
        limit $offset, $limit
